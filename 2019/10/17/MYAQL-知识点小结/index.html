#使用 TRUNCATE 删除全部数据
注意：TRUNCATE TABLE 命令不能用于有外键约束的表，
如果表中存在外键约束则必须使用DELETE命令来删除数据
（实际工作中：不建议使用TRUNCATE命令，
因为使用它删除的数据不能恢复和还原）

#数据查询入门

use mysql;

#SELECT全表查询 查询命令中使用号，将会返回查询的表中所有列
#（如果不使用符号，要查询全表的数据时就需要将全表的列名写出，用逗号分隔）
#SELECT *FROM goods;

#SELECT选择列查询
#注意：一般情况下，最好不用使用*符号进行全表查询
#（虽节省时间，但获取不需要的数据会降低查询和所使用的应用程序的效率）
#SELECT goodsid,goodsing FROM goods;

/*
查询中额别名
使用AS为字段名（列名）定义别名
（AS不是给表里的字段别名，而是给查询的结果字段取别名
所取得别名只限于这一次查询）
/
/*SELECT goodsid AS ‘商品ID’,goodsing AS ‘商品状态’,
goodsname AS ‘商品名’ ,goodpec AS ‘商品进价’,
goodsnumber AS ‘商品数量’,goodsprice AS ‘商品价格’,
goodslate AS ‘商品是否选中’ FROM goods;/
#使用空格为字段名（列名）定义别名
/*SELECT goodsid ‘商品ID’,goodsing ‘商品状态’,
goodsname ‘商品名’ ,goodpec ‘商品进价’,
goodsnumber ‘商品数量’,goodsprice ‘商品价格’,
goodslate ‘商品是否选中’ FROM goods;
*/

#条件查询
#单条件查询
/(查询goodsnumber为NULL，不能用”WHERE goodsnumber=NULL”,
正确表达式为：”goodsnumber IS NULL”)/
#SELECT goodsname,goodsnumber FROM goods WHERE goodsnumber>1;

#多条件复合条件查询 （not，and，or）
#SELECT goodsname,goodsnumber FROM goods WHERE goodsnumber>1 AND goodsname=’帽子’;

#使用distinct消除重复行
#SELECT DISTINCT goodsid,goodsname FROM goods;

#使用LIMIT限定查询返回行
/*LIMIT 位置偏移量，行数；
位置偏移量从0开始（可选参数），从位置偏移量的后一位返回
指定返回的记录条数
*/
#SELECT goodsid,goodsname,goodsnumber FROM goods LIMIT 0,2;

#单列排序 （ASC升序,DESC降序，NULL值比所有值都要小）
#SELECT goodsid,goodsname FROM goods ORDER BY goodsid DESC;

#多列排序 （ASC升序,DESC降序，NULL值比所有值都要小）
/* 例：
SELECT ID，姓名，性别，年龄，出生时间 FROM 表名 ORDER BY 年龄 DESC,出生时间 ASC;
*/

内置函数

日期函数

#获得当前日期+时间（date+time）
#SELECT NOW();
#获得当前日期（date）
#SELECT CURDATE();
#获得当前时间（time）
#SELECT CURTIME();

#返回日期对应的月份（数字类型）
#SELECT MONTH(NOW());
#返回日期对应的月份（字符串）
#SELECT MONTHNAME(NOW());

#返回日期对应的工作日的英文名
#SELECT DAYNAME(NOW());

#返回日期对应的一周中的索引值。1表示周日，2表示周一…
#SELECT DAYOFWEEK(NOW());

#返回日期对应的工作日索引。0表示周日，1表示周一…
#SELECT WEEKDAY(NOW());

#计算日期是一年中的第几周
#SELECT WEEK(NOW());

#计算日期是一年中的第几周
#SELECT WEEKOFYEAR(NOW());

#计算日期是一年中的第几天
#SELECT DAYOFYEAR(NOW());

#计算日期是一个月中的第几天
#SELECT DAYOFMONTH(NOW());
